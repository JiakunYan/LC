145  LCII_context_t* rdv_ctx,
146  bool is_progress)
147 {
148  LCM_DBG_Assert(rdv_ctx->data.lbuffer.address == NULL ||
149  rdv_ctx->data.lbuffer.length >= packet->data.rts.size,

151  "posted by recvl (%lu)!\n",
152  packet->data.rts.size, rdv_ctx->data.lbuffer.length);
153  rdv_ctx->rank = packet->context.src_rank;
154  rdv_ctx->data.lbuffer.length = packet->data.rts.size;
155 
156  LCII_context_t* rtr_ctx = LCIU_malloc(sizeof(LCII_context_t));
157  rtr_ctx->data.mbuffer.address = &(packet->data);
158  LCII_initilize_comp_attr(rtr_ctx->comp_attr);
159  LCII_comp_attr_set_msg_type(rtr_ctx->comp_attr, LCI_MSG_RTR);
160  LCII_comp_attr_set_free_packet(rtr_ctx->comp_attr, 1);
161 
162  packet->context.poolid = -1;
163  uint64_t ctx_key;
164  int result = LCM_archive_put(ep->ctx_archive_p, (uintptr_t)rdv_ctx, &ctx_key);
165  // TODO: be able to pass back pressure to user
166  LCM_Assert(result == LCM_SUCCESS, "Archive is full!\n");
167  packet->data.rtr.recv_ctx_key = ctx_key;
168  if (rdv_ctx->data.lbuffer.address == NULL) {
169  LCI_lbuffer_alloc(ep->device, packet->data.rts.size,
170  &rdv_ctx->data.lbuffer);
171  }
172  if (rdv_ctx->data.lbuffer.segment == LCI_SEGMENT_ALL) {
173  LCM_DBG_Assert(LCII_comp_attr_get_dereg(rdv_ctx->comp_attr) == 1, "\n");
174  LCI_memory_register(ep->device, rdv_ctx->data.lbuffer.address,
175  rdv_ctx->data.lbuffer.length,
176  &rdv_ctx->data.lbuffer.segment);
177  } else {
178  LCM_DBG_Assert(LCII_comp_attr_get_dereg(rdv_ctx->comp_attr) == 0, "\n");
179  }
180  packet->data.rtr.remote_addr_base =
181  (uintptr_t)rdv_ctx->data.lbuffer.segment->mr.address;
182  packet->data.rtr.remote_addr_offset =
183  (uintptr_t)rdv_ctx->data.lbuffer.address -
184  packet->data.rtr.remote_addr_base;
185  packet->data.rtr.rkey = LCIS_rma_rkey(rdv_ctx->data.lbuffer.segment->mr);
186 
187  LCIS_endpoint_t endpoint_to_use;
188  if (is_progress) {
189  endpoint_to_use = ep->device->endpoint_progress.endpoint;
190  } else {

194  packet->data.address, sizeof(struct LCII_packet_rtr_t),
195  ep->device->heap.segment->mr,
