4 // 32 bits for rank, 2 bits for msg type, 14 bits for endpoint ID, 16 bits for
5 // tag
6 static inline uint64_t LCII_make_key(LCI_endpoint_t ep, int rank, LCI_tag_t tag,
7  LCI_msg_type_t msg_type)
8 {
9  uint64_t ret = 0;
10  if (ep->match_type == LCI_MATCH_RANKTAG) {
11  ret = (uint64_t)(rank) << 32 | (uint64_t)(msg_type) << 30 |
12  (uint64_t)(ep->gid) << 16 | (uint64_t)(tag);
13  } else {
14  LCM_DBG_Assert(ep->match_type == LCI_MATCH_TAG, "Unknown match_type %d\n",
15  ep->match_type);
16  ret = (uint64_t)(-1) << 32 | (uint64_t)(msg_type) << 30 |
17  (uint64_t)(ep->gid) << 16 | (uint64_t)(tag);
18  }
19  return ret;
20 }

22 static inline void lc_ce_dispatch(LCII_context_t* ctx)
23 {
24  if (LCII_comp_attr_get_dereg(ctx->comp_attr) == 1) {
25  LCI_memory_deregister(&ctx->data.lbuffer.segment);
26  ctx->data.lbuffer.segment = LCI_SEGMENT_ALL;
27  }
28  switch (LCII_comp_attr_get_comp_type(ctx->comp_attr)) {
29  case LCI_COMPLETION_NONE:
30  LCIU_free(ctx);
31  break;
32 #ifdef LCI_SERVER_HAS_SYNC
33  case LCI_COMPLETION_SYNC: {
34  LCII_sync_signal(ctx->completion, ctx);
35  break;
36  }
37 #endif
38 #ifdef LCI_SERVER_HAS_CQ
39  case LCI_COMPLETION_QUEUE:
40  LCII_queue_push(ctx->completion, ctx);
41  break;
42 #endif
43 #ifdef LCI_SERVER_HAS_AM
44  case LCI_COMPLETION_HANDLER: {
45  LCI_handler_t handler = ctx->completion;
46  (*handler)(LCII_ctx2req(ctx));
47  LCIU_free(ctx);
48  break;
49  }
50 #endif
51  default:
52  LCM_DBG_Assert(false, "Unknown completion type: %d!\n",
53  (int)LCII_comp_attr_get_comp_type(ctx->comp_attr));
54  }
55 }

57 static inline void LCIS_serve_recv(void* p, int src_rank, size_t length,
58  uint32_t imm_data)
59 {
60  LCII_PCOUNTERS_WRAPPER(LCII_pcounters[LCIU_get_thread_id()].msgs_rx += 1);
61  LCII_PCOUNTERS_WRAPPER(LCII_pcounters[LCIU_get_thread_id()].bytes_rx +=
62  length);
63  LCII_PCOUNTERS_WRAPPER(LCII_pcounters[LCIU_get_thread_id()].msgs_2sided_rx +=
64  1);
65  LCII_packet_t* packet = (LCII_packet_t*)p;
66  LCII_proto_t proto = imm_data;
67  // NOTE: this should be RGID because it is received from remote.
68  LCI_endpoint_t ep = LCI_ENDPOINTS[PROTO_GET_RGID(proto)];
69  LCI_tag_t tag = PROTO_GET_TAG(proto);
70  LCI_msg_type_t msg_type = PROTO_GET_TYPE(proto);
71  packet->context.src_rank = src_rank;
72 
73  switch (msg_type) {
74  case LCI_MSG_SHORT: {
75  LCM_DBG_Assert(length == LCI_SHORT_SIZE,
76  "Unexpected message length %lu\n", length);
77  LCM_hashtable_key key = LCII_make_key(ep, src_rank, tag, LCI_MSG_SHORT);
78  LCM_hashtable_val value = (LCM_hashtable_val)packet;
79  if (!LCM_hashtable_insert(ep->mt, key, &value, SERVER)) {
80  LCII_context_t* ctx = (LCII_context_t*)value;
81  // If the receiver uses LCI_MATCH_TAG, we have to set the rank here.
82  ctx->rank = src_rank;
83  memcpy(&(ctx->data.immediate), packet->data.address, LCI_SHORT_SIZE);
84  LCII_free_packet(packet);
85  lc_ce_dispatch(ctx);
86  }
87  break;
88  }
89  case LCI_MSG_MEDIUM: {
90  LCM_hashtable_key key = LCII_make_key(ep, src_rank, tag, LCI_MSG_MEDIUM);
91  LCM_hashtable_val value = (LCM_hashtable_val)packet;
92  packet->context.length = length;
93  if (!LCM_hashtable_insert(ep->mt, key, &value, SERVER)) {
94  LCII_context_t* ctx = (LCII_context_t*)value;
95  ctx->rank = src_rank;
96  ctx->data.mbuffer.length = length;
97  if (ctx->data.mbuffer.address != NULL) {
98  // copy to user provided buffer
99  memcpy(ctx->data.mbuffer.address, packet->data.address,
100  ctx->data.mbuffer.length);
101  LCII_free_packet(packet);
102  } else {
103  // use LCI packet
104  ctx->data.mbuffer.address = packet->data.address;
105  }
106  lc_ce_dispatch(ctx);
107  }
108  break;
109  }
110  case LCI_MSG_RTS: {
111  switch (packet->data.rts.msg_type) {
112  case LCI_MSG_LONG: {
113  const LCM_hashtable_key key =
114  LCII_make_key(ep, src_rank, tag, LCI_MSG_LONG);
115  LCM_hashtable_val value = (LCM_hashtable_val)packet;
116  if (!LCM_hashtable_insert(ep->mt, key, &value, SERVER)) {
117  LCII_handle_2sided_rts(ep, packet, (LCI_comp_t)value, true);
118  }
119  break;
120  }
121  case LCI_MSG_RDMA_LONG: {
122  LCII_handle_1sided_rts(ep, packet, src_rank, tag);
123  break;
124  }
125  case LCI_MSG_IOVEC: {
126  LCII_handle_iovec_rts(ep, packet, src_rank, tag);
127  break;
128  }
129  default:
130  LCM_Assert(false, "Unknown message type %d!\n",
131  packet->data.rts.msg_type);
132  }
133  break;
134  }
135  case LCI_MSG_RTR:
136  switch (packet->data.rts.msg_type) {
137  case LCI_MSG_LONG: {
138  LCII_handle_2sided_rtr(ep, packet);
139  break;
140  }
141  case LCI_MSG_RDMA_LONG: {
142  LCII_handle_1sided_rtr(ep, packet);
143  break;
144  }
145  case LCI_MSG_IOVEC: {
146  LCII_handle_iovec_rtr(ep, packet);
147  break;
148  }
149  default:
150  LCM_Assert(false, "Unknown message type %d!\n",
151  packet->data.rts.msg_type);
152  }
153  break;
154  case LCI_MSG_RDMA_SHORT: {
155  // dynamic put
156  LCM_DBG_Assert(length == LCI_SHORT_SIZE,
157  "Unexpected message length %lu\n", length);
158  LCII_context_t* ctx = LCIU_malloc(sizeof(LCII_context_t));
159  memcpy(&(ctx->data.immediate), packet->data.address, LCI_SHORT_SIZE);
160  LCII_free_packet(packet);
161  ctx->data_type = LCI_IMMEDIATE;
162  ctx->rank = src_rank;
163  ctx->tag = tag;
164  LCII_initilize_comp_attr(ctx->comp_attr);
165  LCII_comp_attr_set_comp_type(ctx->comp_attr, LCI_COMPLETION_QUEUE);
166  ctx->completion = ep->default_comp;
167  ctx->user_context = NULL;
168  lc_ce_dispatch(ctx);
169  break;
170  }
171  case LCI_MSG_RDMA_MEDIUM: {
172  LCII_context_t* ctx = LCIU_malloc(sizeof(LCII_context_t));
173  ctx->data.mbuffer.address = packet->data.address;
174  ctx->data.mbuffer.length = length;
175  ctx->data_type = LCI_MEDIUM;
176  ctx->rank = src_rank;
177  ctx->tag = tag;
178  LCII_initilize_comp_attr(ctx->comp_attr);
179  LCII_comp_attr_set_comp_type(ctx->comp_attr, LCI_COMPLETION_QUEUE);
180  ctx->completion = ep->default_comp;
181  ctx->user_context = NULL;
182  lc_ce_dispatch(ctx);
183  break;
184  }
185  case LCI_MSG_FIN: {
186  LCM_DBG_Assert(length == sizeof(LCII_context_t*),
187  "Unexpected FIN message length (%lu)!\n", length);
188  LCII_handle_iovec_recv_FIN(packet);
189  break;
190  }
191  default:
192  LCM_Assert(false, "Unknown proto!\n");
193  }
