127 static inline int LCISD_poll_cq(LCIS_endpoint_t endpoint_pp,
128  LCIS_cq_entry_t* entry)
129 {
130  LCISI_endpoint_t* endpoint_p = (LCISI_endpoint_t*)endpoint_pp;
131  struct ibv_wc wc[LCI_CQ_MAX_POLL];
132  int ne = ibv_poll_cq(endpoint_p->cq, LCI_CQ_MAX_POLL, wc);
133  LCM_DBG_Assert(ne >= 0, "ibv_poll_cq returns error %d\n", ne);
134 
135  for (int i = 0; i < ne; i++) {
136  LCM_DBG_Assert(
137  wc[i].status == IBV_WC_SUCCESS, "Failed status %s (%d) for wr_id %d\n",
138  ibv_wc_status_str(wc[i].status), wc[i].status, (int)wc[i].wr_id);
139  if (wc[i].opcode == IBV_WC_RECV) {
140  entry[i].opcode = LCII_OP_RECV;
141  entry[i].ctx = (void*)wc[i].wr_id;
142  entry[i].length = wc[i].byte_len;
143  entry[i].imm_data = wc[i].imm_data;
144  entry[i].rank =
145  endpoint_p->qp2rank[wc[i].qp_num % endpoint_p->qp2rank_mod];
146  } else if (wc[i].opcode == IBV_WC_RECV_RDMA_WITH_IMM) {
147  entry[i].opcode = LCII_OP_RDMA_WRITE;
148  entry[i].ctx = (void*)wc[i].wr_id;
149  entry[i].imm_data = wc[i].imm_data;
150  } else {
151  LCM_DBG_Assert(
152  wc[i].opcode == IBV_WC_SEND || wc[i].opcode == IBV_WC_RDMA_WRITE,
153  "Unexpected IBV opcode!\n");
154  entry[i].opcode = LCII_OP_SEND;
155  entry[i].ctx = (void*)wc[i].wr_id;
156  }
157  }
158  return ne;
159 }
