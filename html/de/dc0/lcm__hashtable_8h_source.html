48 static inline uint32_t myhash(const uint64_t k)
49 {
50  uint32_t hash = ((k & 0xff) ^ Seed) * Prime;
51  hash = (((k >> 8) & 0xff) ^ hash) * Prime;
52  hash = (((k >> 16) & 0xff) ^ hash) * Prime;
53  hash = (((k >> 24) & 0xff) ^ hash) * Prime;
54  hash = (((k >> 32) & 0xff) ^ hash) * Prime;
55  hash = (((k >> 40) & 0xff) ^ hash) * Prime;
56  hash = (((k >> 48) & 0xff) ^ hash) * Prime;
57  hash = (((k >> 56) & 0xff) ^ hash) * Prime;
58

60  return (((hash >> LCM_HASHTABLE_BIT_SIZE) ^ hash) &
61  TINY_MASK(LCM_HASHTABLE_BIT_SIZE));

112  // If the key is the same as tag, someone has inserted it.
113  // If the type is different, meaning we can't use it.
114  if (tag == cmp_key) {

116  hentry->entry.tag = LCM_HASHTABLE_EMPTY;
117  LCIU_release_spinlock(&master->control.lock);

120  return 0;
121  } else if (tag == LCM_HASHTABLE_EMPTY) {
122  // Ortherwise, if the tag is empty, we record the slot.
123  // We can't return until we go over all entries.
124  if (empty_hentry == NULL) empty_hentry = hentry;
125  } else {
126  // If we are still seeing some non-empty,
127  // push that empty entry even further.
128  empty_hentry = NULL;
129  }

135  // Moving on to the next.
136  // *** SLOWISH ***
137  if (hcontrol->control.next == NULL) {

141  hcontrol->control.next = LCMI_hashtable_create_table(1);
142  hcontrol = hcontrol->control.next;
143  empty_hentry = hcontrol + 1;
144  }
145  break;

149  hentry = hcontrol + 1;
