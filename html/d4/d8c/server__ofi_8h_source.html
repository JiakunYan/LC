48 static inline void* LCISI_real_server_reg(LCIS_server_t s, void* buf,
49  size_t size)
50 {
51  LCISI_server_t* server = (LCISI_server_t*)s;
52  int rdma_key;
53  if (server->info->domain_attr->mr_mode & FI_MR_PROV_KEY) {
54  rdma_key = 0;
55  } else {
56  rdma_key = __sync_fetch_and_add(&g_next_rdma_key, 1);
57  }
58  struct fid_mr* mr;
59  FI_SAFECALL(fi_mr_reg(server->domain, buf, size,
60  FI_READ | FI_WRITE | FI_REMOTE_WRITE, 0, rdma_key, 0,
61  &mr, 0));
62  if (server->info->domain_attr->mr_mode & FI_MR_ENDPOINT) {
63  LCM_DBG_Assert(server->endpoint_count >= 1, "No endpoints available!\n");
64  if (strcmp(server->info->fabric_attr->prov_name, "cxi") == 0) {
65  // A temporary fix for the cxi provider. It appears cxi cannot bind a
66  // memory region to more than one endpoint, but other endpoints can still
67  // use this memory region to send and recv messages.
68  FI_SAFECALL(fi_mr_bind(mr, &server->endpoints[0]->ep->fid, 0));
69  } else {
70  for (int i = 0; i < server->endpoint_count; ++i) {
71  FI_SAFECALL(fi_mr_bind(mr, &server->endpoints[i]->ep->fid, 0));
72  }
73  }
74  FI_SAFECALL(fi_mr_enable(mr));
75  }
76  return (void*)mr;
77 }

109 static inline int LCISD_poll_cq(LCIS_endpoint_t endpoint_pp,
110  LCIS_cq_entry_t* entry)
111 {
112  LCISI_endpoint_t* endpoint_p = (LCISI_endpoint_t*)endpoint_pp;
113  struct fi_cq_data_entry fi_entry[LCI_CQ_MAX_POLL];
114  struct fi_cq_err_entry error;
115  ssize_t ne;
116  int ret;
117 
118  ne = fi_cq_read(endpoint_p->cq, &fi_entry, LCI_CQ_MAX_POLL);
119  ret = ne;
120  if (ne > 0) {
121  // Got an entry here
122  for (int i = 0; i < ne; i++) {
123  if (fi_entry[i].flags & FI_RECV) {
124  entry[i].opcode = LCII_OP_RECV;
125  entry[i].ctx = fi_entry[i].op_context;
126  entry[i].length = fi_entry[i].len;
127  entry[i].imm_data = fi_entry[i].data & ((1ULL << 32) - 1);
128  entry[i].rank = (int)(fi_entry[i].data >> 32);
129  } else if (fi_entry[i].flags & FI_REMOTE_WRITE) {
130  entry[i].opcode = LCII_OP_RDMA_WRITE;
131  entry[i].ctx = NULL;
132  entry[i].imm_data = fi_entry[i].data;
133  } else {
134  LCM_DBG_Assert(
135  fi_entry[i].flags & FI_SEND || fi_entry[i].flags & FI_WRITE,
136  "Unexpected OFI opcode!\n");
137  entry[i].opcode = LCII_OP_SEND;
138  entry[i].ctx = fi_entry[i].op_context;
139  }
140  }
141  } else if (ne == -FI_EAGAIN) {
142  ret = 0;
143  } else {
144  LCM_DBG_Assert(ne == -FI_EAVAIL, "unexpected return error: %s\n",
145  fi_strerror(-ne));
146  fi_cq_readerr(endpoint_p->cq, &error, 0);
147  LCM_Assert(false, "Err %d: %s\n", error.err, fi_strerror(error.err));
148  }
149  return ret;
150 }
